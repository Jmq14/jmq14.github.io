Basic Calculator

问题描述：

Implement a basic calculator to evaluate a simple expression string.

The expression string may contain open ( and closing parentheses ), the plus + or minus sign -, non-negative integers and empty spaces .

You may assume that the given expression is always valid.

这道题与作业题非常像， 作业要求的方法为利用栈将中缀表达式转换为后缀表达式，然后再通过后缀表达式顺序扫描计算出结果。
大致修改一下作业代码之后在LeetCode上submit了一下，关键代码如下：

void inFixToPostFix(string &inFix, string &postFix)
{
    stack<char> st;
    char c;
    int isAfterNum = 0;
    int len = inFix.length();

    for (int i = 0; i < len; i++){
        c = inFix[i];
        if (c == '('){
            isAfterNum = 0;
            st.push(c);
        }
        else if (c == ')'){
            isAfterNum = 0;
            while (st.top() != '('){
                postFix.append(1, st.top());
                st.pop();
            }
            st.pop();
        }
        else{
            if (!IsOperator(c)){
                if (isAfterNum) st.top() = (st.top() - '0')* 10 + c;
                else st.push(c);
                isAfterNum = 1;
            }
            else{
                isAfterNum = 0;
                while (!st.empty() && OperatorPrecedence(st.top(), c) >= 0){
                    postFix.append(1, st.top());
                    st.pop();
                }
                st.push(c);
            }
        }
    }
    while (!st.empty()){
        postFix.append(1, st.top());
        st.pop();
    }
}

int calculatePostFix(string &postFix)
{
    int tmp = 0;
    char c;
    stack<int> st;
    int len = postFix.length();
    int num1 = 0, num2 = 0;

    for (int i = 0; i< len; i++){
        c = postFix[i];
        if (IsOperator(c)){
            // when c is an operator, extract two top numbers to calculate and then push back the result.
            num1 = st.top();
            st.pop();
            num2 = st.top();
            st.pop();
            switch (c){
            case '+': tmp = num2 + num1; break;
            case '-': tmp = num2 - num1; break;
            case '*': tmp = num2 * num1; break;
            case '/': tmp = num2 / num1; break;
            }
            st.push(tmp);
        }
        else{
            st.push(c-'0');
        }
    }

    //the stack leaves one element number and that is the answer.
    tmp = st.top();
    st.pop();

    return tmp;
}

int calculate(string s) 
{
    string expression;
    inFixToPostFix(s, expression);
    int ans = calculatePostFix(expression);
    return ans;
}

虽然习题平台上这样做AC了，但是在这种做法在LeetCode上评测时存在很多问题：

  1. 后缀表达式用字符串表示，难以处理多位数的运算。作业题中我在转成后缀表达式中运用char型尝试存储了多位数，但是能存的数字大小非常有限。事实上，LeetCode上有样例中包含了数字INT_MAX。
  2. 这道题说明了只需处理加减运算，这种算法没有根据这个要求进行优化。
  3. 先开了一个栈把中缀改成后缀（字符串），再开一个栈计算结果，效率很低。这道题只要求了计算结果，所以不必要一定求出后缀表达式。

根据以上问题，首先就是要分开存储数字和运算符号，所以我想到用两个类型不同的栈分别存储。另外，由于没有乘除运算等优先度更高的运算，所以在扫描时只需要在遇到’(‘和’)’时处理优先级即可。扫描一遍，同时进行压栈存储和出栈计算，所以时间复杂度和空间复杂度均为O(n)。

关键代码如下：

int calculate(string s)
{
    stack<int> nums;   //数字栈
    stack<char> ops;   //运算符栈
    int len = s.length();
    bool afterNumber = false;   // 用于判断是否为多位数

    for (int i = 0; i < len; i++){
        if (s[i] == ' ') continue;

        //遇到左括号直接压栈
        if (s[i] == '('){
            ops.push(s[i]);
            afterNumber = false;
        }
        //遇到右括号则将括号内所有内容出栈计算
        else if (s[i] == ')'){
            afterNumber = false;
            while (ops.top() != '('){
                num1 = nums.top();
                nums.pop();
                num2 = nums.top();
                nums.pop();
                tmpCh = ops.top();
                ops.pop();
                nums.push(addOrMinus(num1, num2, tmpCh));
            }
            ops.pop();
        }
        //遇到运算符
        else if (isOperator(s[i])){
            afterNumber = false;
            //在运算符栈非空且该运算符优先级小于或等于栈顶运算符时将栈顶运算符取出进行运算
            //在这个问题中，只要栈顶不是左括号就都可以进行计算了。
            if ((!ops.empty()) && (ops.top() != '(')){
                num1 = nums.top();
                nums.pop();
                num2 = nums.top();
                nums.pop();
                tmpCh = ops.top();
                ops.pop();
                nums.push(addOrMinus(num1, num2, tmpCh));
            }
            ops.push(s[i]);
        }
        //遇到数字
        else{
            if (afterNumber){
                num1 = nums.top() * 10 + s[i] - '0';
                nums.pop();
                nums.push(num1);
            }
            else{
                nums.push(s[i] - '0');
            }
            afterNumber = true;
        }
    }

    //扫描完成后把两个栈内所有元素取出进行运算，最后数字栈的栈顶元素即为运算答案
    while (!ops.empty()){
        num1 = nums.top();
        nums.pop();
        num2 = nums.top();
        nums.pop();
        tmpCh = ops.top();
        ops.pop();
        nums.push(addOrMinus(num1, num2, tmpCh));
    }
    return nums.top();
}

这种算法在OJ上AC了，但是Runtime不算是比较小的（Runtime 48ms, beats 35.35% of cpp submissions），所以我还是比较不爽（233）。
LeetCode自己给了个Hint是Binary Tree，但是我觉得二叉树跟我用两个栈没太大区别，甚至会更麻烦（吧）。我觉得算法再进行优化的话，肯定是针对因为只有加减运算来的，且复杂度已经降到理论上最小值了，所以肯定是从细节上有所改变。
再读一遍题，注意到一个细节，就是题目给出的数字是non-negative integers，也就是说第一个数字肯定是正数（括号里的第一个数字也是）。那么可不可以不要用一个字符型的栈来存储符号，直接用+1，-1来表示加和减运算，由于不用判断负数，所以这个操作相对很简便，且省略了调用函数来进行二元运算的过程。 另外，在没有遇到括号的时候，由于一直是顺序运算，所以没有必要压栈，直接进行累积运算，遇到括号之后保留括号前的运算结果即可。

关键代码如下：

int calculate(string s)
{
    stack<int> nums;
    stack<int> ops;
    int len = s.length();
    bool afterNumber = false;
    char c;
    int tmpOp = 1, tmpNum = 0, ans = 0;
    int num1 = 0, num2 = 0;

    for (int i = 0; i < len; i++){
        c = s[i];
        if (c == ' ') continue;
        if (c >= '0' && c <= '9'){
            tmpNum = tmpNum*10 + c - '0';
        }
        else{
            ans += tmpNum * tmpOp;
            tmpNum = 0;
            if (c == '('){
                ops.push(tmpOp);
                nums.push(ans);
                tmpOp = 1;
                ans = 0;
            }
            else if (c == ')' && !ops.empty()){
                ans = nums.top() + ans * ops.top();
                nums.pop();
                ops.pop();
            }
            else if (c == '+'){    
                tmpOp = 1;
            }
            else if (c == '-'){    
                tmpOp = -1;
            }
        }
    }
    ans += tmpNum * tmpOp;
    return ans;
}

代码变得非常简洁啊(´∇｀) Runtime:16ms, beats 92.49% of cpp submissions. 当然这之中还有一点插曲，我本来按照上一次提交的代码来修改，提交后运行时间是24ms，后来感觉写了太多重复代码，所以就调整了一下次序（先判断是否是数字，如果遇到的是字符就统一做了前一次的二元运算再来区分这一次的运算符是什么），以及在每一次循环中用一个字符c代替s[i]（虽然没有验证这个对提高效率的贡献大不大），然后运行时间一下子就减少了很多呢！（其实只减少了一点点(´∇｀) ，但是排名上升了不少233）

个人体会就是，确实要根据题目的独特属性来思考算法，以及代码的整洁性等等细节对效率也有着重要的影响，要尽量避免重复的工作。
